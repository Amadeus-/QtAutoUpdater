127 bool UpdateController::start(DisplayLevel displayLevel)
128 {
129  if(d->running)
130  return false;
131  d->running = true;
132  emit runningChanged(true);
133  d->wasCanceled = false;
134  d->displayLevel = displayLevel;
135 
136  if(d->displayLevel >= AskLevel) {
137  if(DialogMaster::questionT(d->window,
138  tr("Check for Updates"),
139  tr("Do you want to check for updates now?"))
140  != QMessageBox::Yes) {
141  d->running = false;
142  emit runningChanged(false);
143  return false;
144  }
145  }
146 
147  if(!d->mainUpdater->checkForUpdates()) {
148  if(d->displayLevel >= ProgressLevel) {
149  DialogMaster::warningT(d->window,
150  tr("Check for Updates"),
151  tr("The program is already checking for updates!"));
152  }
153  d->running = false;
154  emit runningChanged(false);
155  return false;
156  } else {
157  if(d->displayLevel >= ExtendedInfoLevel) {
158  d->checkUpdatesProgress = new ProgressDialog(d->window);
159  if(d->displayLevel >= ProgressLevel) {
160  connect(d->checkUpdatesProgress.data(), &ProgressDialog::canceled, this, [this](){
161  d->wasCanceled = true;
162  });
163  d->checkUpdatesProgress->open(d->mainUpdater, &QtAutoUpdater::Updater::abortUpdateCheck);
164  }
165  }
166 
167  return true;
168  }
169 }

202 void UpdateController::checkUpdatesDone(bool hasUpdates, bool hasError)
203 {
204  if(d->displayLevel >= ExtendedInfoLevel) {
205  auto iconType = QMessageBox::NoIcon;
206  if(hasUpdates)
207  iconType = QMessageBox::Information;
208  else {
209  if(d->wasCanceled || !d->mainUpdater->exitedNormally())
210  iconType = QMessageBox::Warning;
211  else
212  iconType = QMessageBox::Critical;
213  }
214 
215  if(d->checkUpdatesProgress) {
216  d->checkUpdatesProgress->hide(iconType);
217  d->checkUpdatesProgress->deleteLater();
218  d->checkUpdatesProgress = nullptr;
219  }
220  }
221 
222  if(d->wasCanceled) {
223  if(d->displayLevel >= ExtendedInfoLevel) {
224  DialogMaster::warningT(d->window,
225  tr("Check for Updates"),
226  tr("Checking for updates was canceled!"));
227  }
228  } else {
229  if(hasUpdates) {
230  if(d->displayLevel >= InfoLevel) {
231  auto shouldShutDown = false;
232  const auto oldRunAdmin = d->runAdmin;
233  const auto res = UpdateInfoDialog::showUpdateInfo(d->mainUpdater->updateInfo(),
234  d->runAdmin,
235  d->adminUserEdit,
236  d->detailedInfo,
237  d->window);
238  if(d->runAdmin != oldRunAdmin)
239  emit runAsAdminChanged(d->runAdmin);
240 
241  switch(res) {
242  case UpdateInfoDialog::InstallNow:
243  shouldShutDown = true;
244  case UpdateInfoDialog::InstallLater:
245  d->mainUpdater->runUpdaterOnExit(d->runAdmin ? new AdminAuthorization() : nullptr);
246  if(shouldShutDown)
247  qApp->quit();
248  case UpdateInfoDialog::NoInstall:
249  break;
250  default:
251  Q_UNREACHABLE();
252  }
253  } else {
254  d->mainUpdater->runUpdaterOnExit(d->runAdmin ? new AdminAuthorization() : nullptr);
255  if(d->displayLevel == ExitLevel) {
256  DialogMaster::informationT(d->window,
257  tr("Install Updates"),
258  tr("New updates are available. The maintenance tool will be "
259  "started to install those as soon as you close the application!"));
260  } else
261  qApp->quit();
262  }
263  } else {
264  if(hasError) {
265  qCWarning(logQtAutoUpdater) << "maintenancetool process finished with exit code"
266  << d->mainUpdater->errorCode()
267  << "and error string:"
268  << d->mainUpdater->errorLog();
269  }
270 
271  if(d->displayLevel >= ExtendedInfoLevel) {
272  if(d->mainUpdater->exitedNormally()) {
273  DialogMaster::criticalT(d->window,
274  tr("Check for Updates"),
275  tr("No new updates available!"));
276  } else {
277  DialogMaster::warningT(d->window,
278  tr("Check for Updates"),
279  tr("The update process crashed!"));
280  }
281  }
282  }
283  }
284 
285  d->running = false;
286  emit runningChanged(false);
287 }
