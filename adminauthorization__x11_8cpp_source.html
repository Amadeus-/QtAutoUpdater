91 static bool execAdminFallback(const QString &program, const QStringList &arguments)
92 {
93  // as we cannot pipe the password to su in QProcess, we need to setup a pseudo-terminal for it
94  int masterFD = -1;
95  int slaveFD = -1;
96  char ptsn[ PATH_MAX ];
97 
98  if (::openpty(&masterFD, &slaveFD, ptsn, 0, 0))
99  return false;
100 
101  masterFD = ::posix_openpt(O_RDWR | O_NOCTTY);
102  if (masterFD < 0)
103  return false;
104 
105  const QByteArray ttyName = ::ptsname(masterFD);
106 
107  if (::grantpt(masterFD)) {
108  ::close(masterFD);
109  return false;
110  }
111 
112  ::revoke(ttyName);
113  ::unlockpt(masterFD);
114 
115  slaveFD = ::open(ttyName, O_RDWR | O_NOCTTY);
116  if (slaveFD < 0) {
117  ::close(masterFD);
118  return false;
119  }
120 
121  ::fcntl(masterFD, F_SETFD, FD_CLOEXEC);
122  ::fcntl(slaveFD, F_SETFD, FD_CLOEXEC);
123  int pipedData[2];
124  if (pipe(pipedData) != 0)
125  return false;
126 
127  int flags = ::fcntl(pipedData[0], F_GETFL);
128  if (flags != -1)
129  ::fcntl(pipedData[0], F_SETFL, flags | O_NONBLOCK);
130 
131  flags = ::fcntl(masterFD, F_GETFL);
132  if (flags != -1)
133  ::fcntl(masterFD, F_SETFL, flags | O_NONBLOCK);
134 
135  pid_t child = fork();
136 
137  if (child < -1) {
138  ::close(masterFD);
139  ::close(slaveFD);
140  ::close(pipedData[0]);
141  ::close(pipedData[1]);
142  return false;
143  }
144 
145  // parent process
146  else if (child > 0) {
147  ::close(slaveFD);
148  //close writing end of pipe
149  ::close(pipedData[1]);
150 
151  QRegExp re(QLatin1String("[Pp]assword.*:"));
152  QByteArray data;
153  QByteArray errData;
154  int bytes = 0;
155  int errBytes = 0;
156  char buf[1024];
157  char errBuf[1024];
158  while (bytes >= 0) {
159  int state;
160  if (::waitpid(child, &state, WNOHANG) == -1){
161  break;
162  }
163  bytes = ::read(masterFD, buf, 1023);
164  if (bytes == -1 && errno == EAGAIN)
165  bytes = 0;
166  else if (bytes > 0){
167  if(!QByteArray(buf, bytes).simplified().isEmpty())
168  data.append(buf, bytes);
169  else
170  bytes = 0;
171  }
172  errBytes = ::read(pipedData[0], errBuf, 1023);
173  if (errBytes > 0)
174  {
175  errData.append(errBuf, errBytes);
176  errBytes=0;
177  }
178  if (bytes > 0) {
179  const QString line = QString::fromLatin1(data);
180  if (re.indexIn(line) != -1) {
181  if(!errData.isEmpty()) {
182  DialogMaster::critical(nullptr, QString::fromLocal8Bit(errData));
183  errData.clear();
184  }
185 
186  bool ok = false;
187  const QString password = QInputDialog::getText(nullptr,
188  QCoreApplication::translate("AdminAuthorization", "Enter Password"),
189  QCoreApplication::translate("AdminAuthorization", "Enter your root password to run the maintenancetool:"),
190  QLineEdit::Password,
191  QString(),
192  &ok,
193  Qt::WindowCloseButtonHint | Qt::WindowStaysOnTopHint);
194 
195  if (!ok) {
196  QByteArray pwd = password.toLatin1();
197  for (int i = 0; i < 3; ++i) {
198  ::write(masterFD, pwd.data(), pwd.length());
199  ::write(masterFD, "\n", 1);
200  }
201  return false;
202  }
203  QByteArray pwd = password.toLatin1();
204  ::write(masterFD, pwd.data(), pwd.length());
205  ::write(masterFD, "\n", 1);
206  ::read(masterFD, buf, pwd.length() + 1);
207  }
208  }
209  if (bytes == 0)
210  ::usleep(100000);
211  }
212  if (!errData.isEmpty()) {
213  DialogMaster::critical(nullptr, QString::fromLocal8Bit(errData));
214  return false;
215  }
216 
217  int status;
218  child = ::wait(&status);
219  ::close(pipedData[1]);
220  return true;
221  }
222 
223  // child process
224  else {
225  ::close(pipedData[0]);
226  // Reset signal handlers
227  for (int sig = 1; sig < NSIG; ++sig)
228  signal(sig, SIG_DFL);
229  signal(SIGHUP, SIG_IGN);
230 
231  ::setsid();
232 
233  ::ioctl(slaveFD, TIOCSCTTY, 1);
234  int pgrp = ::getpid();
235  ::tcsetpgrp(slaveFD, pgrp);
236 
237  ::dup2(slaveFD, 0);
238  ::dup2(slaveFD, 1);
239  ::dup2(pipedData[1], 2);
240 
241  // close all file descriptors
242  struct rlimit rlp;
243  getrlimit(RLIMIT_NOFILE, &rlp);
244  for (int i = 3; i < static_cast<int>(rlp.rlim_cur); ++i)
245  ::close(i);
246 
247  char **argp = (char **) ::malloc(arguments.count() + 4 * sizeof(char *));
248  QList<QByteArray> args;
249  args.push_back(SU_COMMAND);
250  args.push_back("-b");
251  args.push_back(program.toLocal8Bit());
252  for (QStringList::const_iterator it = arguments.begin(); it != arguments.end(); ++it)
253  args.push_back(it->toLocal8Bit());
254 
255  int i = 0;
256  for (QList<QByteArray>::iterator it = args.begin(); it != args.end(); ++it, ++i)
257  argp[i] = it->data();
258  argp[i] = 0;
259 
260  ::unsetenv("LANG");
261  ::unsetenv("LC_ALL");
262 
263  ::execv(SU_COMMAND, argp);
264  _exit(0);
265  return false;
266  }
267 }
