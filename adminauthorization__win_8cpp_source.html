81 bool AdminAuthorization::executeAsAdmin(const QString &program, const QStringList &arguments)
82 {
83  DeCoInitializer _;
84 
85  // AdminAuthorization::execute uses UAC to ask for admin privileges. If the user is no
86  // administrator yet and the computer's policies are set to not use UAC (which is the case
87  // in some corporate networks), the call to execute() will simply succeed and not at all
88  // launch the child process. To avoid this, we detect this situation here and return early.
89  if (!hasAdminRights()) {
90  QLatin1String key("HKEY_LOCAL_MACHINE\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Policies\\System");
91  QSettings registry(key, QSettings::NativeFormat);
92  const QVariant enableLUA = registry.value(QLatin1String("EnableLUA"));
93  if ((enableLUA.type() == QVariant::Int) && (enableLUA.toInt() == 0))
94  return false;
95  }
96 
97  const QString file = QDir::toNativeSeparators(program);
98  const QString args = qt_create_commandline(arguments);
99 
100  SHELLEXECUTEINFOW shellExecuteInfo;
101  ZeroMemory(&shellExecuteInfo, sizeof(SHELLEXECUTEINFOW));
102  shellExecuteInfo.cbSize = sizeof(SHELLEXECUTEINFOW);
103  shellExecuteInfo.lpVerb = L"runas";
104  shellExecuteInfo.lpFile = (wchar_t *)file.utf16();
105  shellExecuteInfo.lpParameters = (wchar_t *)args.utf16();
106  shellExecuteInfo.nShow = SW_SHOW;
107 
108  if (ShellExecuteExW(&shellExecuteInfo))
109  return true;
110  else
111  return false;
112 }

114 QString qt_create_commandline(const QStringList &arguments)
115 {
116  QString args;
117  for (int i = 0; i < arguments.size(); ++i) {
118  QString tmp = arguments.at(i);
119  // in the case of \" already being in the string the \ must also be escaped
120  tmp.replace(QLatin1String("\\\""), QLatin1String("\\\\\""));
121  // escape a single " because the arguments will be parsed
122  tmp.replace(QLatin1Char('\"'), QLatin1String("\\\""));
123  if (tmp.isEmpty() || tmp.contains(QLatin1Char(' ')) || tmp.contains(QLatin1Char('\t'))) {
124  // The argument must not end with a \ since this would be interpreted
125  // as escaping the quote -- rather put the \ behind the quote: e.g.
126  // rather use "foo"\ than "foo\"
127  QString endQuote(QLatin1Char('\"'));
128  int i = tmp.length();
129  while (i > 0 && tmp.at(i - 1) == QLatin1Char('\\')) {
130  --i;
131  endQuote += QLatin1Char('\\');
132  }
133  args += QLatin1String(" \"") + tmp.left(i) + endQuote;
134  } else {
135  args += QLatin1Char(' ') + tmp;
136  }
137  }
138  return args;
139 }
